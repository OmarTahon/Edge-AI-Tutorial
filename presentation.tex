\section{Introduction to Edge AI}

\begin{frame}{What is Edge AI?}
    \textbf{Edge AI} moves machine learning inference from centralized cloud servers directly to local hardware devices.
    
    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Why move to the Edge?}
        \begin{itemize}
            \item \textbf{Latency:} Eliminate round-trip delays for real-time applications (e.g., autonomous driving).
            \item \textbf{Bandwidth:} Avoid streaming massive amounts of raw data (like video feeds).
            \item \textbf{Privacy:} Sensitive data remains on-device.
        \end{itemize}
        \item \textbf{What is TinyML?} 
        \begin{itemize}
            \item A subset of Edge AI focused on deploying models on ultra-low power Microcontrollers (MCUs) running on milliwatts of power.
        \end{itemize}
    \end{itemize}
\end{frame}

\section{Model Compression \& Quantization}

\begin{frame}{Model Quantization}
    \textbf{Problem:} Floating-point (FP32) weights consume too much memory and power for edge devices.
    
    \vspace{0.3cm}
    \textbf{Solution:} \textbf{Quantization} maps high-precision floats to lower precision integers (INT8, INT4).
    
    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Post-Training Quantization (PTQ):} Convert after training. Fast, but can lose accuracy.
        \item \textbf{Quantization-Aware Training (QAT):} Simulate low precision during training. Recovers lost accuracy.
        \item \textbf{AWQ (Activation-aware Weight Quantization):} Only quantizes non-salient (non-critical) weights, allowing 4-bit precision with minimal accuracy loss.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Code Example: INT8 Quantization}
    \begin{lstlisting}
import numpy as np

def quantize_int8(weights):
    min_val, max_val = np.min(weights), np.max(weights)
    
    # Calculate scale and zero-point for INT8 [-128, 127]
    scale = (max_val - min_val) / 255.0
    zero_point = -128 - np.round(min_val / scale)
    
    # Quantize and clip
    q_weights = np.round(weights / scale + zero_point)
    q_weights = np.clip(q_weights, -128, 127).astype(np.int8)
    
    return q_weights, scale, zero_point
    \end{lstlisting}
    \small \textit{Result: A 4.0x compression ratio (FP32 $\rightarrow$ INT8) reducing memory and increasing inference speed.}
\end{frame}
