% ================= SECTION 1: Introduction ================= %
\section{Introduction to Edge AI}

\begin{frame}{The Edge Paradigm}
    \textbf{Problem:} Traditional Machine Learning relies heavily on centralized cloud servers. This approach introduces significant bottlenecks:
    
    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{High Latency:} Network round-trips prevent real-time autonomous decision-making.
        \item \textbf{Bandwidth Constraints:} Streaming continuous high-resolution sensor data (e.g., 4K video) is expensive and often impossible.
        \item \textbf{Privacy Concerns:} Sending sensitive biometric or personal data to the cloud poses security risks.
    \end{itemize}
    
    \vspace{0.3cm}
    \textbf{Solution:} \textbf{Edge AI} executes inference locally on hardware accelerators (NPUs, TPUs) or Microcontrollers (TinyML).
\end{frame}

% ================= SECTION 2: SOTA Architectures ================= %
\section{State-of-the-Art Architectures}

\begin{frame}{Vision at the Edge: YOLOv11 \& MobileNetV4}
    To run on constrained devices, models must be aggressively optimized without losing accuracy.
    
    \vspace{0.3cm}
    \textbf{YOLOv11 (November 2024)}
    \begin{itemize}
        \item Replaces traditional Non-Maximum Suppression (NMS) with dual label assignment, creating a fully NMS-free end-to-end pipeline.
        \item \textbf{Result:} 25-40\% reduction in latency compared to YOLOv10, achieving 60+ FPS on edge processors.
    \end{itemize}
    
    \vspace{0.3cm}
    \textbf{MobileNetV4}
    \begin{itemize}
        \item Utilizes the \textbf{Universal Inverted Bottleneck (UIB)}, dynamically searching for the best macro-architecture using Neural Architecture Search (NAS).
        \item \textbf{Result:} 87\% ImageNet accuracy at just 3.8ms latency on Edge TPUs.
    \end{itemize}
\end{frame}

% ================= SECTION 3: Quantization ================= %
\section{Model Compression \& Quantization}

\begin{frame}{The Math of Quantization}
    \textbf{Quantization} maps 32-bit floating-point (FP32) weights to 8-bit or 4-bit integers (INT8/INT4).
    
    \vspace{0.3cm}
    The affine quantization mapping is defined as:
    \begin{equation}
        q = \text{round}\left(\frac{r}{S} + Z\right)
    \end{equation}
    where:
    \begin{itemize}
        \item $r$: The real FP32 value.
        \item $S$: The strictly positive \textit{Scale} factor.
        \item $Z$: The integer \textit{Zero-point} (offset), ensuring $r=0$ maps exactly to an integer.
        \item $q$: The resulting quantized $n$-bit integer.
    \end{itemize}
\end{frame}

\begin{frame}{Schematic: FP32 to INT8 Mapping}
    \begin{center}
        \begin{tikzpicture}
            % Continuous line
            \draw[thick, <->] (-4, 0) -- (4, 0) node[right] {FP32 Range $[-r_{max}, r_{max}]$};
            
            % Discrete line
            \draw[thick, <->] (-4, -2) -- (4, -2) node[right] {INT8 Range $[-128, 127]$};
            
            % Mapping arrows
            \draw[arrow, dashed] (-3, -0.2) -- (-3, -1.8) node[midway, left] {Quantize};
            \draw[arrow, dashed] (3, -0.2) -- (3, -1.8);
            \draw[arrow, dashed, black] (0, -0.2) -- (0, -1.8) node[midway, right] {Zero Point ($Z$)};
            
            % Ticks
            \foreach \x in {-3,-1.5,0,1.5,3}
                \draw (\x, 0.2) -- (\x, -0.2);
            \foreach \x in {-3,-2.5,...,3}
                \draw (\x, -1.8) -- (\x, -2.2);
                
        \end{tikzpicture}
    \end{center}
    \vspace{0.3cm}
    \small \textit{Post-Training Quantization (PTQ) establishes $S$ and $Z$ after training, while Quantization-Aware Training (QAT) simulates this discrete mapping during the backward pass using Straight-Through Estimators (STE).}
\end{frame}

\begin{frame}[fragile]{Code Example: FP32 Inference Performance}
    \begin{lstlisting}
import numpy as np
import time

# Simulate large feature maps (e.g., 4K image embeddings)
N = 5000
weights_fp32 = np.random.randn(N, N).astype(np.float32)
activations_fp32 = np.random.randn(N, N).astype(np.float32)

# FP32 Inference
start = time.time()
out_fp32 = np.dot(weights_fp32, activations_fp32)
print(f"FP32 Runtime: {time.time() - start:.4f} s") # ~0.8500 s
    \end{lstlisting}
\end{frame}

\section{Conclusion}

\begin{frame}{Summary}
    \begin{itemize}
        \item \textbf{Edge AI} shifts intelligence from the cloud to the device, solving latency, bandwidth, and privacy issues.
        \item Efficient architectures (e.g., \textbf{MobileNetV4}, \textbf{YOLOv11}) use advanced techniques like Depthwise Convolutions and PSA to minimize parameter counts.
        \item \textbf{Quantization} (FP32 $\rightarrow$ INT8/INT4) dramatically reduces memory footprints and leverages highly optimized integer math units, speeding up inference by up to 7x.
        \item Dedicated toolchains like \textbf{TI \texttt{edgeai-tensorlab}} abstract away complex NPU compilation, creating a seamless path from dataset to physical device deployment.
    \end{itemize}
\end{frame}
